生成指定班级课程表，可视化版

-- 生成指定班级课程表，可视化版
select
	arrangement_section as 节次,
	group_concat(case when arrangement_week = '一' then course_name end) as 一,
	group_concat(case when arrangement_week = '二' then course_name end) as 二,
	group_concat(case when arrangement_week = '三' then course_name end) as 三,
	group_concat(case when arrangement_week = '四' then course_name end) as 四,
	group_concat(case when arrangement_week = '五' then course_name end) as 五,
	group_concat(case when arrangement_week = '六' then course_name end) as 六,
	group_concat(case when arrangement_week = '日' then course_name end) as 日
from arrangement
join course on course.course_id = arrangement.course_id
join class1 on class1.class_id = arrangement.class_id
where class_name like '2023多媒体制作班'
group by arrangement_section
order by arrangement_section

生成指定教师课程表，带班级、班主任，可视化版，可显示冲突课程

-- 生成指定教师课程表，带班级、班主任，可视化版
with RECURSIVE numbers as (
	select 1 as n
	union all
	select n+1 from numbers where n < 10
)
select
n as 节次,
一,二,三,四,五,六,日
from numbers left outer join
(select
arrangement_section,
group_concat(case when arrangement_week = '一' then concat(课程名,' 【', 班级名, '_', teacher_name, '】') end) as 一,
group_concat(case when arrangement_week = '二' then concat(课程名,' 【', 班级名, '_', teacher_name, '】') end) as 二,
group_concat(case when arrangement_week = '三' then concat(课程名,' 【', 班级名, '_', teacher_name, '】') end) as 三,
group_concat(case when arrangement_week = '四' then concat(课程名,' 【', 班级名, '_', teacher_name, '】') end) as 四,
group_concat(case when arrangement_week = '五' then concat(课程名,' 【', 班级名, '_', teacher_name, '】') end) as 五,
group_concat(case when arrangement_week = '六' then concat(课程名,' 【', 班级名, '_', teacher_name, '】') end) as 六,
group_concat(case when arrangement_week = '日' then concat(课程名,' 【', 班级名, '_', teacher_name, '】') end) as 日
from (select
	class_name as 班级名,
	class1.teacher_id as 班主任,
	teacher.teacher_name as 任课教师,
	course.course_name as 课程名,
	arrangement_week,
	arrangement_section
from arrangement
join course on course.course_id = arrangement.course_id
join class1 on class1.class_id = arrangement.class_id
join teacher on teacher.teacher_id = arrangement.teacher_id
where 任课教师 like '陈颖超' ) table1
join teacher on teacher.teacher_id = table1.班主任
group by arrangement_section
order by arrangement_section) table2
on numbers.n = table2.arrangement_section

显示所有班级课表，班级名，星期，节次，课程名，教师名

-- 显示所有班级课表，班级名，星期，节次，课程名，教师名
select
	class_name as '班级名',
	arrangement_week as '星期',
	arrangement_section as '节次' ,
	course_name as '课程名',
	teacher_name as '教师名'
from arrangement
join class1 on arrangement.class_id = class1.class_id
join course on arrangement.course_id = course.course_id
join classroom on arrangement.classroom_id = classroom.classroom_id
join teacher on arrangement.teacher_id = teacher.teacher_id
order by
	class_name asc,
	case arrangement_week
		when '一' then 1
		when '二' then 2
		when '三' then 3
		when '四' then 4
		when '五' then 5
		when '六' then 6
		when '日' then 7
		else 999
	end asc,
	arrangement_section asc

显示姓名、班级、学科、周总课时数

select
	姓名,
	group_concat(distinct 班级名) as 班级名,
	group_concat(distinct 课程名) as 课程名,
	sum(课时数) as 总课时数
from
	(select
		teacher_name as 姓名,
		class_name as 班级名,
		course_name as 课程名,
		count(*) as 课时数
	from arrangement
	join course on course.course_id = arrangement.course_id
	join class1 on class1.class_id = arrangement.class_id
	join teacher on teacher.teacher_id = arrangement.teacher_id
	where teacher_name in
	('刘帅', '李佳展',
	'李连花', '张雪莹',
	'杨晴', '陈颖超', '符恺璐',
	'孟久轩', '王艳会', '冯云鹤',
	'牛俊芝', '杨硕')
	group by teacher_name, class_name, course_name
	order by 姓名 desc) table1
group by 姓名
order by 总课时数 desc

显示姓名、任教年级、任教学科、周课时（学校教职工分工明细表）

-- 显示 姓名，任教年级，任教学科，周课时（学校教职工分工明细表）
select 姓名, 
concat('信息工程系 ', 年级) as 任教年级,
课程 as 任教课程, 课时数
from 
(select 姓名,
group_concat(年级, '、') as 年级,
group_concat(课程,'、') as 课程, 
sum(课时数) as 课时数 
from 
(select 
姓名,
年级,
group_concat(课程,'、') as 课程, 
sum(课时数) as 课时数
from 
(select 
teacher_name as 姓名, 
case 
	when strftime('%m', date('now')) in ("09", "10", "11", "12") then
		case strftime('%Y', date('now'))-substr(class_name, 1, 4)
			when 0 then "职一"
			when 1 then "职二"
			when 2 then "职三"
		end
	else 
	    case strftime('%Y', date('now'))-substr(class_name, 1, 4)
			when 1 then "职一"
			when 2 then "职二"
			when 3 then "职三"
		end
end as 年级,
course_name as 课程,
count(*) as 课时数 
from arrangement
join class1 on arrangement.class_id = class1.class_id
join course on arrangement.course_id = course.course_id
join classroom on arrangement.classroom_id = classroom.classroom_id
join teacher on arrangement.teacher_id = teacher.teacher_id
group by 姓名,年级,课程) 
as table1
group by 姓名,年级)
as table2
group by 姓名)
as table3

生成 专业课课时统计表【not in】 或 公共课课时统计表【in】

-- 生成 专业课课时统计表【not in】 或 公共课课时统计表【in】
select
	姓名,
	sum(课时数) as 总课时数,
	group_concat(课时详情, '、') as 课时详情
from
(select
	teacher_name as 姓名,
	concat(class_name, "_", course_name, "(", count(*), ")") as 课时详情,
	count(*) as 课时数
from arrangement
join course on course.course_id = arrangement.course_id
join class1 on class1.class_id = arrangement.class_id
join teacher on teacher.teacher_id = arrangement.teacher_id
where course_name not in
	("语文", "数学", "英语", "体育",
	"德育", "劳动", "历史", "美育", "美术",
	"信息技术", "音乐", "数字技术应用")
group by teacher_name, class_name, course_name
order by 姓名 desc) table1
group by 姓名
order by 总课时数 desc
